async function sendBLEData() { if (!bleChar) return; const w = canvas.width; const h = canvas.height; const bufferSize = (w * h) / 8; // 15,000 Bytes cho màn 400x300 log("Đang trích xuất dữ liệu ảnh (Đen & Đỏ)..."); const imgData = ctx.getImageData(0, 0, w, h); const d = imgData.data; const bufBlack = new Uint8Array(bufferSize); const bufRed = new Uint8Array(bufferSize); // Mặc định phủ trắng 2 mảng (màu trắng trong E-paper 1-bit là 1 / 0xFF) bufBlack.fill(0xFF); bufRed.fill(0xFF); // Chuyển đổi dữ liệu Pixel từ Canvas sang dạng Bitmap 1-bit for (let y = 0; y < h; y++) { for (let x = 0; x < w; x++) { let pixelIdx = (y * w + x) * 4; let r = d[pixelIdx]; let g = d[pixelIdx + 1]; let b = d[pixelIdx + 2]; let byteIdx = Math.floor((y * w + x) / 8); let bitIdx = 7 - ((y * w + x) % 8); // Nhận diện màu Đỏ (R cao, G và B thấp) if (r > 200 && g < 100 && b < 100) { // Màu Đỏ: Trong mảng Red bit = 0 bufRed[byteIdx] &= ~(1 << bitIdx); } // Nhận diện màu Đen (Cả R, G, B đều thấp) else if (r < 100 && g < 100 && b < 100) { // Màu Đen: Trong mảng Black bit = 0 bufBlack[byteIdx] &= ~(1 << bitIdx); } } } // Ghép 2 mảng bufBlack và bufRed lại để gửi liên tục xuống ESP32 const fullPayload = new Uint8Array(bufferSize * 2); fullPayload.set(bufBlack, 0); fullPayload.set(bufRed, bufferSize); log(`Đang truyền gói dữ liệu (${fullPayload.length} bytes) qua BLE...`); // Chia nhỏ gửi theo Chunk 512 bytes const CHUNK_SIZE = 512; for (let i = 0; i < fullPayload.length; i += CHUNK_SIZE) { const chunk = fullPayload.slice(i, i + CHUNK_SIZE); await bleChar.writeValueWithoutResponse(chunk); await new Promise(r => setTimeout(r, 20)); // Delay nhẹ 20ms tránh nghẽn BLE } log("✅ Đã gửi xong! Màn hình đang tiến hành làm tươi (Refresh)..."); }
